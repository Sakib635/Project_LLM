(declare-fun future () String)
(declare-fun pymel () String)
(assert (let ((a!1 (or (= future "1.0.0")
               (= future "0.18.3")
               (= future "0.18.2")
               (= future "0.18.1")
               (= future "0.18.0")
               (= future "0.9.0")
               (= future "0.8.2")
               (= future "0.8.1")
               (= future "0.8.0")
               (= future "0.7.0")
               (= future "0.6.0")
               (= future "0.5.2")
               (= future "0.5.1")
               (= future "0.5.0")
               (= future "0.4.1")
               (= future "0.4.0")
               (= future "0.3.5")
               (= future "0.3.4")
               (= future "0.3.3")
               (= future "0.3.2")
               (= future "0.3.1")
               (= future "0.3.0")
               (= future "0.2.0"))))
  (and (or (= pymel "1.2.0")
           (= pymel "1.2.0a8")
           (= pymel "1.2.0a6")
           (= pymel "1.2.0a5"))
       (=> (= pymel "1.2.0") a!1)
       (=> (= pymel "1.2.0a8") a!1)
       (=> (= pymel "1.2.0a6") a!1)
       (=> (= pymel "1.2.0a5") a!1))))
(check-sat)
