(declare-fun six () String)
(declare-fun protobuf () String)
(declare-fun sgmllib3k () String)
(declare-fun feedparser () String)
(assert (and (or (= feedparser "6.0.2"))
     (or (= protobuf "3.14.0")
         (= protobuf "3.14.0rc3")
         (= protobuf "3.14.0rc2")
         (= protobuf "3.14.0rc1"))
     (=> (= feedparser "6.0.2") (or (= sgmllib3k "1.0.0")))
     (=> (= protobuf "3.14.0") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc3") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc2") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc1") (or (= six "1.9.0")))))
(check-sat)
