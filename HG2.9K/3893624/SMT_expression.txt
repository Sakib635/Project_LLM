(declare-fun six () String)
(declare-fun protobuf () String)
(declare-fun MarkupSafe () String)
(declare-fun Jinja2 () String)
(declare-fun webapp2 () String)
(assert (and (or (= Jinja2 "2.11.2"))
     (or (= protobuf "3.14.0")
         (= protobuf "3.14.0rc3")
         (= protobuf "3.14.0rc2")
         (= protobuf "3.14.0rc1"))
     (or (= webapp2 "2.5.2"))
     (=> (= Jinja2 "2.11.2")
         (or (= MarkupSafe "2.1.5")
             (= MarkupSafe "2.1.4")
             (= MarkupSafe "2.1.3")
             (= MarkupSafe "2.1.2")
             (= MarkupSafe "2.1.1")
             (= MarkupSafe "2.1.0")
             (= MarkupSafe "2.0.1")
             (= MarkupSafe "2.0.0")
             (= MarkupSafe "2.0.0rc2")
             (= MarkupSafe "2.0.0rc1")
             (= MarkupSafe "2.0.0a1")
             (= MarkupSafe "1.1.1")
             (= MarkupSafe "1.1.0")
             (= MarkupSafe "1.0")
             (= MarkupSafe "0.23")
             (= MarkupSafe "0.9.3")
             (= MarkupSafe "0.9.2")
             (= MarkupSafe "0.9.1")
             (= MarkupSafe "0.9")))
     (=> (= protobuf "3.14.0") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc3") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc2") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc1") (or (= six "1.9.0")))))
(check-sat)
