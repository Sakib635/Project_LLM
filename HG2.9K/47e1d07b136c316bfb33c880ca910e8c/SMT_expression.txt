(declare-fun mccabe () String)
(declare-fun flake8 () String)
(declare-fun pycodestyle () String)
(declare-fun pyflakes () String)
(assert (and (or (= flake8 "3.8.4"))
     (=> (= flake8 "3.8.4") (or (= pyflakes "2.2.0")))
     (=> (= flake8 "3.8.4")
         (or (= pycodestyle "2.6.0") (= pycodestyle "2.6.0a1")))
     (=> (= flake8 "3.8.4") (or (= mccabe "0.6.1") (= mccabe "0.6.0")))))
(check-sat)
