(declare-fun mechanize () String)
(declare-fun webencodings () String)
(declare-fun six () String)
(assert (let ((a!1 (or (= webencodings "0.5.1")
               (= webencodings "0.5")
               (= webencodings "0.4")
               (= webencodings "0.3")
               (= webencodings "0.2")
               (= webencodings "0.1"))))
  (and (or (= mechanize "0.4.5"))
       (=> (= html5lib "1.1") (or (= six "1.9.0")))
       (=> (= html5lib "1.1") a!1)
       (=> (= html5lib "1.0.1") (or (= six "1.9.0")))
       (=> (= html5lib "1.0.1") a!1)
       (=> (= mechanize "0.4.5")
           (or (= html5lib "1.1")
               (= html5lib "1.0.1")
               (= html5lib "1.0")
               (= html5lib "1.0b10")
               (= html5lib "1.0b9")
               (= html5lib "1.0b8")
               (= html5lib "1.0b7")
               (= html5lib "1.0b6")
               (= html5lib "1.0b5")
               (= html5lib "1.0b3")
               (= html5lib "1.0b2")
               (= html5lib "1.0b1")
               (= html5lib "0.999999999"))))))
