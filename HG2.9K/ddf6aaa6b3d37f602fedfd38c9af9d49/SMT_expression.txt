(declare-fun six () String)
(declare-fun protobuf () String)
(declare-fun youtube_dl () String)
(assert (and (or (= protobuf "3.14.0")
         (= protobuf "3.14.0rc3")
         (= protobuf "3.14.0rc2")
         (= protobuf "3.14.0rc1"))
     (or (= youtube_dl "2021.1.8"))
     (=> (= protobuf "3.14.0") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc3") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc2") (or (= six "1.9.0")))
     (=> (= protobuf "3.14.0rc1") (or (= six "1.9.0")))))
(check-sat)
